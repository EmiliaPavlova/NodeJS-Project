Create a Postgres docker container:
docker run --name node-db-postgres -e POSTGRES_PASSWORD=123456 -d postgres

docker ps
docker start node-db-postgres
docker stop [CONTAINER ID]

docker exec -it node-db-postgres psql -U postgres

List containers
docker container ls --all


Connect to DB:
postgres=#
CREATE DATABASE nodejsmentoringprogram;
\c nodejsmentoringprogram

nodejsmentoringprogram=#
CREATE TABLE Products (
    product_id integer UNIQUE PRIMARY KEY NOT NULL,
    name varchar(40) NOT NULL,
	price REAL
);

INSERT INTO Products (product_id, name, price) VALUES
    (01, 'Product 1', 10.50),
    (02, 'Product 2', 9.00);


CREATE TABLE Users (
     user_id integer UNIQUE PRIMARY KEY NOT NULL,
     name varchar(40) NOT NULL CHECK (name <> '')
);

INSERT INTO Users (user_id, name) VALUES
    (1, 'Ivan Ivanov'),
    (2, 'Anton Peev'),
    (3, 'Ana Ilieva),
    (4, 'Harry Potter');
	
\q

Automate:
docker run --name nodejsmentoringprogram -v "$PWD"/:/app -e POSTGRES_PASWORD=123456 -d postgres
docker exec -it nodejsmentoringprogram psql -U postgres -c "CREATE DATABASE node_db;"
docker exec -it nodejsmentoringprogram psql -U postgres -f /script_node.sql
